cmake_minimum_required(VERSION 2.8)
PROJECT(dmtimer)

SET( CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -std=c++11" )

IF (WIN32)  
    MESSAGE(STATUS "Now is windows")  
ELSEIF (APPLE)  
    MESSAGE(STATUS "Now is Apple systems.")  
    ADD_DEFINITIONS(-DDMTIMER_USE_MACOSX)
ELSEIF (UNIX)  
    MESSAGE(STATUS "Now is UNIX-like OS's. Including Apple os x and CygWin")  
    ADD_DEFINITIONS(-DDMTIMER_USE_LINUX)
ENDIF ()

IF (WIN32)
    INCLUDE_DIRECTORIES(${CMAKE_CURRENT_SOURCE_DIR}/)
    INCLUDE_DIRECTORIES(${CMAKE_CURRENT_SOURCE_DIR}/include)

    LINK_DIRECTORIES(${CMAKE_CURRENT_SOURCE_DIR}/lib/win)

    SET(EXECUTABLE_OUTPUT_PATH ${CMAKE_CURRENT_SOURCE_DIR}/bin)
    SET(LIBRARY_OUTPUT_PATH ${CMAKE_CURRENT_SOURCE_DIR}/lib/win)

    FILE(GLOB_RECURSE DMTIMER_SOURCES
        ${CMAKE_CURRENT_SOURCE_DIR}/include/*.cpp
        ${CMAKE_CURRENT_SOURCE_DIR}/include/*.cc
        ${CMAKE_CURRENT_SOURCE_DIR}/include/*.c
        ${CMAKE_CURRENT_SOURCE_DIR}/include/*.hpp
        ${CMAKE_CURRENT_SOURCE_DIR}/include/*.h

        ${CMAKE_CURRENT_SOURCE_DIR}/src/*.cpp
        ${CMAKE_CURRENT_SOURCE_DIR}/src/*.cc
        ${CMAKE_CURRENT_SOURCE_DIR}/src/*.c
        ${CMAKE_CURRENT_SOURCE_DIR}/src/*.hpp
        ${CMAKE_CURRENT_SOURCE_DIR}/src/*.h)

    ADD_LIBRARY(dmtimer ${DMTIMER_SOURCES})

    FILE(GLOB_RECURSE DMTIMERTEST_SOURCES
    ${CMAKE_CURRENT_SOURCE_DIR}/test/*.cpp
    ${CMAKE_CURRENT_SOURCE_DIR}/test/*.cc
    ${CMAKE_CURRENT_SOURCE_DIR}/test/*.c
    ${CMAKE_CURRENT_SOURCE_DIR}/test/*.hpp
    ${CMAKE_CURRENT_SOURCE_DIR}/test/*.h)

    ADD_EXECUTABLE(dmtimertest ${DMTIMERTEST_SOURCES})

    TARGET_LINK_LIBRARIES(dmtimertest dmtimer)
ELSEIF (APPLE)
    INCLUDE_DIRECTORIES(${CMAKE_CURRENT_SOURCE_DIR}/)
    INCLUDE_DIRECTORIES(${CMAKE_CURRENT_SOURCE_DIR}/include)

    LINK_DIRECTORIES(${CMAKE_CURRENT_SOURCE_DIR}/lib/mac)

    SET(EXECUTABLE_OUTPUT_PATH ${CMAKE_CURRENT_SOURCE_DIR}/bin)
    SET(LIBRARY_OUTPUT_PATH ${CMAKE_CURRENT_SOURCE_DIR}/lib/mac)

    FILE(GLOB_RECURSE DMTIMER_SOURCES
        ${CMAKE_CURRENT_SOURCE_DIR}/include/*.cpp
        ${CMAKE_CURRENT_SOURCE_DIR}/include/*.cc
        ${CMAKE_CURRENT_SOURCE_DIR}/include/*.c
        ${CMAKE_CURRENT_SOURCE_DIR}/include/*.hpp
        ${CMAKE_CURRENT_SOURCE_DIR}/include/*.h

        ${CMAKE_CURRENT_SOURCE_DIR}/src/*.cpp
        ${CMAKE_CURRENT_SOURCE_DIR}/src/*.cc
        ${CMAKE_CURRENT_SOURCE_DIR}/src/*.c
        ${CMAKE_CURRENT_SOURCE_DIR}/src/*.hpp
        ${CMAKE_CURRENT_SOURCE_DIR}/src/*.h)

    ADD_LIBRARY(dmtimer ${DMTIMER_SOURCES})

    FILE(GLOB_RECURSE TEST_SOURCES
    ${CMAKE_CURRENT_SOURCE_DIR}/test/*.cpp
    ${CMAKE_CURRENT_SOURCE_DIR}/test/*.cc
    ${CMAKE_CURRENT_SOURCE_DIR}/test/*.c
    ${CMAKE_CURRENT_SOURCE_DIR}/test/*.hpp
    ${CMAKE_CURRENT_SOURCE_DIR}/test/*.h)

    ADD_EXECUTABLE(dmtimertest ${TEST_SOURCES})

    TARGET_LINK_LIBRARIES(dmtimertest pthread dmtimer)
ELSE(UNIX)
    INCLUDE_DIRECTORIES(${CMAKE_CURRENT_SOURCE_DIR}/)
    INCLUDE_DIRECTORIES(${CMAKE_CURRENT_SOURCE_DIR}/include)

    LINK_DIRECTORIES(${CMAKE_CURRENT_SOURCE_DIR}/lib/lin)

    SET(EXECUTABLE_OUTPUT_PATH ${CMAKE_CURRENT_SOURCE_DIR}/bin)
    SET(LIBRARY_OUTPUT_PATH ${CMAKE_CURRENT_SOURCE_DIR}/lib/lin)

    FILE(GLOB_RECURSE DMTIMER_SOURCES
        ${CMAKE_CURRENT_SOURCE_DIR}/include/*.cpp
        ${CMAKE_CURRENT_SOURCE_DIR}/include/*.cc
        ${CMAKE_CURRENT_SOURCE_DIR}/include/*.c
        ${CMAKE_CURRENT_SOURCE_DIR}/include/*.hpp
        ${CMAKE_CURRENT_SOURCE_DIR}/include/*.h

        ${CMAKE_CURRENT_SOURCE_DIR}/src/*.cpp
        ${CMAKE_CURRENT_SOURCE_DIR}/src/*.cc
        ${CMAKE_CURRENT_SOURCE_DIR}/src/*.c
        ${CMAKE_CURRENT_SOURCE_DIR}/src/*.hpp
        ${CMAKE_CURRENT_SOURCE_DIR}/src/*.h)

    ADD_LIBRARY(dmtimer ${DMTIMER_SOURCES})

    FILE(GLOB_RECURSE TEST_SOURCES
    ${CMAKE_CURRENT_SOURCE_DIR}/test/*.cpp
    ${CMAKE_CURRENT_SOURCE_DIR}/test/*.cc
    ${CMAKE_CURRENT_SOURCE_DIR}/test/*.c
    ${CMAKE_CURRENT_SOURCE_DIR}/test/*.hpp
    ${CMAKE_CURRENT_SOURCE_DIR}/test/*.h)

    ADD_EXECUTABLE(dmtimertest ${TEST_SOURCES})

    TARGET_LINK_LIBRARIES(dmtimertest rt pthread dmtimer)
ENDIF ()
